jQuery Bing maps polygons and polylines

Using jQuery with Bing maps

Download jQuery 1.4.X or higher or use Googles or Microsofts CDN. 

<script type="text/javascript" src="http://ecn.dev.virtualearth.net/mapcontrol/mapcontrol.ashx?v=7.0"></script>
<script type="text/javascript" src="http://ajax.aspnetcdn.com/ajax/jQuery/jquery-1.7.min.js"></script>
<script type="text/javascript" src="PATH_TO_PLUGIN/jquery.ui.bmap.js"></script>
<script type="text/javascript" src="PATH_TO_PLUGIN/jquery.ui.bmap.overlays.js"></script>
var polygonCoords = ['20,-20', '20,20', '-20,20', '-20,-20', '20,-20'];
var polylinesCoords = ['20,-20', '-20,20'];
$('#map_canvas').gmap({'credentials': YOUR_BING_MAP_CREDENTIALS}).bind('init', function(ev, map) {
	$('#map_canvas').gmap('Polygon', { 
		'paths': polygonCoords, 
		'fillColor': new Microsoft.Maps.Color(100,100,0,100), 
		'strokeColor': new Microsoft.Maps.Color(200,0,100,100),
		'strokeThickness': 5 
	});
	$('#map_canvas').gmap('Polyline', { 
		'paths': polylinesCoords, 
		'strokeColor': new Microsoft.Maps.Color(200, 100, 0, 100), 
		'strokeThickness': 5, 
		'strokeDashArray': '5 2' 
	});
});
$('#map_canvas').gmap({'credentials': YOUR_BING_MAP_CREDENTIALS, 'callback': function() {
	var self = this;
	self.addShape('Polygon', { 
		'paths': polygonCoords, 
		'fillColor': new Microsoft.Maps.Color(100,100,0,100), 
		'strokeColor': new Microsoft.Maps.Color(200,0,100,100),
		'strokeThickness': 5 
	});
	self.addShape('Polyline', { 
		'paths': polylinesCoords, 
		'strokeColor': new Microsoft.Maps.Color(200, 100, 0, 100), 
		'strokeThickness': 5, 
		'strokeDashArray': '5 2' 
	});
}});
$('#map_canvas').gmap('Polygon', { 
	'paths': polygonCoords, 
	'fillColor': new Microsoft.Maps.Color(100,100,0,100), 
	'strokeColor': new Microsoft.Maps.Color(200,0,100,100),
	'strokeThickness': 5 
});
$('#map_canvas').gmap('Polyline', { 
	'paths': polylinesCoords, 
	'strokeColor': new Microsoft.Maps.Color(200, 100, 0, 100), 
	'strokeThickness': 5, 
	'strokeDashArray': '5 2' 
});
